Start

DECLARE Integer i,n,a,b,c
 
ASSIGN a=0 ,b=1 ,c=0 

OUTPUT "Input the number"

INPUT n

ASSIGN i=2

IF n==0    Then ,
            OUTPUT "The number is"&a

                  Else,
             FOR (i=2 ; i<=n ; i++)
               NEXT ,  
                        ASSIGN c = a+b
                        ASSIGN a=b
                        ASSIGN b=c
               DONE , 
                        OUTPUT "The number is"&b
 End
